DIPLOMACY_CREATE_FACTION_ENABLE_TRIGGER = {
	IF = {
		limit = {
			has_game_rule = {
				rule = r56rp_enable_bakedinrules
				option = ENABLED
			}
		}
		IF = { # AI can invite ai
			limit = {
				is_ai = yes
			}
			ROOT = { #sender
				OR = {
					has_war = no
					has_war_together_with = FROM
				}
			}
			FROM = { #recipient
				is_ai = yes
				OR = {
					has_war = no
					has_war_together_with = ROOT
				}
			}
		}
		ELSE_IF = { # Players can invite players
			limit = {
				is_ai = no
			}
			ROOT = { #sender
				OR = {
					has_war = no
					has_war_together_with = FROM
				}
			}
			FROM = { #recipient
				OR = {
					is_subject_of = ROOT
					AND = {
						is_ai = no
						OR = {
							has_war = no
							has_war_together_with = ROOT
						}
					}
				}
			}
		}
	}
}

DIPLOMACY_WAR_ENABLE_TRIGGER = {
	IF = { # this doesent have to be under baked in rules, its host tool anyways.
		limit = {
			OR = {
				has_idea = r56rp_ht_freeze_nation_idea
				FROM = {
					has_idea = r56rp_ht_freeze_nation_idea
				}
			}
		}
		NOT = {
			OR = {
				ROOT = {
					has_idea = r56rp_ht_freeze_nation_idea
				}
				FROM = {
					has_idea = r56rp_ht_freeze_nation_idea
				}
			}
		}
	}
	IF = {
		limit = {
			has_game_rule = {
				rule = r56rp_enable_bakedinrules
				option = ENABLED
			}
		}
		
		IF = {
			limit = {
				check_variable = {
					var = r56rp_rules_fp_controlled
					value = 5
					compare = greater_than_or_equals
				}
			}
			date > 1939.7.1
			#FROM = { #depracated for now AKA cant be bothered
			#	NOT = { has_idea = r56rp_anti_dogpile_rule }
			#}
		}

		ELSE_IF = {
			limit = {
				check_variable = {
					var = r56rp_rules_fp_controlled
					value = 4
					compare = equals
				}
			}

			IF = {
				limit = {
					tag = JAP
					FROM = {
						tag = CHI
					}
				}
				OR = {
					date > 1939.1.1
					AND = {
						has_completed_focus = JAP_intervene_in_china
					}
				}
			}
			ELSE_IF = { #china can declare on 2 warlords.
				limit = {
					tag = CHI
					FROM = {
						OR = {
							tag = GXC
							tag = MEN
							tag = SHX
							tag = XSM
							tag = SIK
							tag = TNG
							tag = YUN
							tag = PRC
						}
						is_ai = yes
					}
				}
				OR = {
					date > 1939.1.1
					custom_trigger_tooltip = {
						tooltip = r56rp_scripdip_1
						NOT = {
							AND = { #theese flags are applied with an on_action when war is declared.
								has_country_flag = r56rp_chi_declared_on_warlord_early_1
								has_country_flag = r56rp_chi_declared_on_warlord_early_2
							}
						}
					}					
				}
			}
			ELSE_IF = { # PRC as a 4 pointer can declare on other warlords
				limit = {
					tag = PRC
					FROM = {
						OR = {
							tag = GXC
							tag = MEN
							tag = SHX
							tag = PRC
							tag = XSM
							tag = SIK
							tag = TNG
							tag = YUN
						}
					}
				}

				OR = {
					date > 1939.1.1
					custom_trigger_tooltip = {
						tooltip = r56rp_scripdip_2
						FROM = {
							OR = {
								tag = GXC
								tag = MEN
								tag = SHX
								tag = PRC
								tag = XSM
								tag = SIK
								tag = TNG
								tag = YUN
							}
						}
					}
				}
			}
			ELSE = {
				date > 1939.1.1
			}
		}

		ELSE_IF = {
			limit = {
				check_variable = {
					var = r56rp_rules_fp_controlled
					value = 3
					compare = equals
				}
			}
			date > 1938.1.1
		}
		ELSE_IF = {
			limit = {
				check_variable = {
					var = r56rp_rules_fp_controlled
					value = 2
					compare = equals
				}
			}
			IF = { #this if for "all warlords can declare on other warlords always"
				limit = {
					tag = GXC
					FROM = {
						OR = {
							tag = GXC
							tag = MEN
							tag = SHX
							tag = PRC
							tag = XSM
							tag = SIK
							tag = TNG
							tag = YUN
						}
					}
				}

				OR = {
					date > 1937.1.1
					custom_trigger_tooltip = {
						tooltip = r56rp_scripdip_2
						FROM = {
							OR = {
								tag = GXC
								tag = MEN
								tag = SHX
								tag = PRC
								tag = XSM
								tag = SIK
								tag = TNG
								tag = YUN
							}
						}
					}
				}
			}
			ELSE = {
				date > 1937.1.1
			}
			
		}
		ELSE_IF = {
			limit = {
				check_variable = {
					var = r56rp_rules_fp_controlled
					value = 1
					compare = equals
				}
			}
			date > 1936.1.1
		}
		ELSE = {
			date > 1937.1.1
		}
	}
}
